Build your first NEAR-enabled React application in just a few steps. This guide will walk you through setting up providers, connecting wallets, fetching account info, and executing transactions.

## Setup Your Provider

<Steps>
  <Step title="Choose Your Network">
    Start by wrapping your app with a provider. Use `TestnetNearProvider` for development or `MainnetNearProvider` for production.

    <CodeGroup>
    ```tsx Testnet
    import { TestnetNearProvider } from 'react-near-ts';

    export default function App() {
      return (
        <TestnetNearProvider>
          <YourApp />
        </TestnetNearProvider>
      );
    }
    ```

    ```tsx Mainnet
    import { MainnetNearProvider } from 'react-near-ts';

    export default function App() {
      return (
        <MainnetNearProvider>
          <YourApp />
        </MainnetNearProvider>
      );
    }
    ```
    </CodeGroup>
  </Step>

  <Step title="Connect a Wallet">
    Use the `useNearConnector` hook to enable wallet connections.

    ```tsx WalletButton.tsx
    import { useNearConnector, useConnectedAccount } from 'react-near-ts';

    export function WalletButton() {
      const { connect, disconnect } = useNearConnector();
      const { connectedAccountId, isConnectedAccount } = useConnectedAccount();

      if (isConnectedAccount) {
        return (
          <div>
            <p>Connected as: {connectedAccountId}</p>
            <button onClick={() => disconnect.mutate()}>
              Disconnect
            </button>
          </div>
        );
      }

      return (
        <button onClick={() => connect.mutate()}>
          Connect Wallet
        </button>
      );
    }
    ```
  </Step>

  <Step title="Fetch Account Balance">
    Retrieve account information using `useAccountInfo`.

    ```tsx AccountBalance.tsx
    import { useAccountInfo, useConnectedAccount } from 'react-near-ts';

    export function AccountBalance() {
      const { connectedAccountId } = useConnectedAccount();
      const accountInfo = useAccountInfo({ accountId: connectedAccountId });

      if (accountInfo.isLoading) return <p>Loading balance...</p>;
      if (accountInfo.isError) return <p>Error loading balance</p>;

      return (
        <div>
          <h3>Account Balance</h3>
          <p>Total: {accountInfo.data.accountInfo.balance.total.near} NEAR</p>
          <p>Available: {accountInfo.data.accountInfo.balance.available.near} NEAR</p>
        </div>
      );
    }
    ```
  </Step>

  <Step title="Send a Transaction">
    Execute transactions using `useExecuteTransaction` with action creators like `transfer` or `functionCall`.

    ```tsx SendNEAR.tsx
    import { useExecuteTransaction, transfer } from 'react-near-ts';
    import { useState } from 'react';

    export function SendNEAR() {
      const executeTransaction = useExecuteTransaction();
      const [receiver, setReceiver] = useState('');
      const [amount, setAmount] = useState('');

      const handleSend = () => {
        executeTransaction.mutate({
          intent: {
            action: transfer({ amount: { near: amount } }),
            receiverAccountId: receiver,
          },
        });
      };

      return (
        <div>
          <input
            placeholder="Receiver account"
            value={receiver}
            onChange={(e) => setReceiver(e.target.value)}
          />
          <input
            placeholder="Amount (NEAR)"
            value={amount}
            onChange={(e) => setAmount(e.target.value)}
          />
          <button
            onClick={handleSend}
            disabled={executeTransaction.isPending}
          >
            {executeTransaction.isPending ? 'Sending...' : 'Send NEAR'}
          </button>
        </div>
      );
    }
    ```
  </Step>
</Steps>

## Call Smart Contracts

### Read Contract Data

Use `useContractReadFunction` to call view-only contract methods.

```tsx ReadContract.tsx
import {
  useContractReadFunction,
  fromJsonBytes,
  type DeserializeResultFnArgs,
} from 'react-near-ts';
import * as z from 'zod';

const ResultSchema = z.array(z.string());

const deserializeResult = ({ rawResult }: DeserializeResultFnArgs) =>
  ResultSchema.parse(fromJsonBytes(rawResult));

export function ContractRecords() {
  const records = useContractReadFunction({
    contractAccountId: 'react-near-ts.lantstool.testnet',
    functionName: 'get_records',
    functionArgs: { author_id: 'example.testnet' },
    withStateAt: 'LatestOptimisticBlock',
    options: { deserializeResult },
  });

  if (records.isLoading) return <p>Loading records...</p>;
  if (records.isError) return <p>Error loading records</p>;

  return (
    <ul>
      {records.data.map((record, i) => (
        <li key={i}>{record}</li>
      ))}
    </ul>
  );
}
```

### Write Contract Data

Execute contract methods that change state using `functionCall`.

```tsx WriteContract.tsx
import { useExecuteTransaction, functionCall } from 'react-near-ts';
import { useState } from 'react';

export function AddRecord() {
  const executeTransaction = useExecuteTransaction();
  const [record, setRecord] = useState('');

  const handleAdd = () => {
    executeTransaction.mutate({
      intent: {
        action: functionCall({
          functionName: 'add_record',
          functionArgs: { record },
          gasLimit: { teraGas: '10' },
        }),
        receiverAccountId: 'react-near-ts.lantstool.testnet',
      },
    });
  };

  return (
    <div>
      <input
        placeholder="Enter record"
        value={record}
        onChange={(e) => setRecord(e.target.value)}
      />
      <button onClick={handleAdd}>
        Add Record
      </button>
    </div>
  );
}
```

## Complete Example

Here's a full working application combining all the pieces:

```tsx App.tsx
import { TestnetNearProvider, useNearConnector, useConnectedAccount, useAccountInfo, useExecuteTransaction, transfer } from 'react-near-ts';

function DashboardContent() {
  const { connect, disconnect } = useNearConnector();
  const { connectedAccountId, isConnectedAccount } = useConnectedAccount();
  const accountInfo = useAccountInfo({ accountId: connectedAccountId });
  const executeTransaction = useExecuteTransaction();

  if (!isConnectedAccount) {
    return <button onClick={() => connect.mutate()}>Connect Wallet</button>;
  }

  return (
    <div>
      <h2>Welcome, {connectedAccountId}</h2>
      
      {accountInfo.isSuccess && (
        <p>Balance: {accountInfo.data.accountInfo.balance.available.near} NEAR</p>
      )}

      <button
        onClick={() => executeTransaction.mutate({
          intent: {
            action: transfer({ amount: { near: '0.01' } }),
            receiverAccountId: 'receiver.testnet',
          },
        })}
      >
        Send 0.01 NEAR
      </button>

      <button onClick={() => disconnect.mutate()}>Disconnect</button>
    </div>
  );
}

export default function App() {
  return (
    <TestnetNearProvider>
      <DashboardContent />
    </TestnetNearProvider>
  );
}
```
