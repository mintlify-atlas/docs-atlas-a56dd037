## Quick Example

Connect a wallet and execute a transaction with just a few lines of code:

```tsx App.tsx
import { MainnetNearProvider, useNearConnector, useExecuteTransaction, transfer } from 'react-near-ts';

function WalletButton() {
  const { connect, disconnect } = useNearConnector();
  const executeTransaction = useExecuteTransaction();

  return (
    <>
      <button onClick={() => connect.mutate()}>Connect Wallet</button>
      <button onClick={() => executeTransaction.mutate({
        intent: {
          action: transfer({ amount: { near: '0.1' } }),
          receiverAccountId: 'receiver.testnet'
        }
      })}>
        Send NEAR
      </button>
    </>
  );
}

export default function App() {
  return (
    <MainnetNearProvider>
      <WalletButton />
    </MainnetNearProvider>
  );
}
```
