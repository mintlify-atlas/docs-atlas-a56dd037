## Overview

`useContractReadFunction` calls read-only smart contract functions on NEAR. It uses TanStack Query for automatic caching, background refetching, and request deduplication. Supports custom serialization/deserialization of arguments and results.

## Import

```typescript
import { useContractReadFunction } from 'react-near-ts';
```

## Parameters

<ParamField path="contractAccountId" type="AccountId" required>
  The account ID of the smart contract to call.
  
  Example: `"contract.testnet"` or `"app.near"`
  
  **Note:** Query will not execute if `contractAccountId` is `undefined`.
</ParamField>

<ParamField path="functionName" type="ContractFunctionName" required>
  The name of the contract function to call.
  
  Example: `"get_records"` or `"get_user_data"`
  
  **Note:** Query will not execute if `functionName` is `undefined`.
</ParamField>

<ParamField path="functionArgs" type="any">
  Arguments to pass to the contract function. Will be JSON-serialized by default unless custom `serializeArgs` is provided.
  
  Example: `{ user_id: "alice.testnet" }`
</ParamField>

<ParamField path="withStateAt" type="BlockReference">
  Block reference to query contract state at a specific point in time.
  
  Options:
  - `"LatestOptimisticBlock"` - Most recent block (default)
  - `"LatestFinalBlock"` - Latest finalized block
  - `{ blockHash: string }` - Specific block by hash
  - `{ blockHeight: number }` - Specific block by height
</ParamField>

<ParamField path="options" type="object">
  Custom serialization and deserialization functions.

    <ParamField path="options.deserializeResult" type="BaseDeserializeResultFn">
      Custom function to deserialize the contract result.
      
      Signature: `(args: { rawResult: Uint8Array }) => T`

</ParamField>

## Usage

### Basic Usage

```typescript
import { useContractReadFunction } from 'react-near-ts';

function ContractData() {
  const result = useContractReadFunction({
    contractAccountId: 'contract.testnet',
    functionName: 'get_status',
  });

  if (result.isPending) return <p>Loading...</p>;
  if (result.isError) return <p>Error: {result.error.message}</p>;

  return <pre>{JSON.stringify(result.data.result, null, 2)}</pre>;
}
```

### With Function Arguments

```typescript
import { useContractReadFunction } from 'react-near-ts';

function UserRecords({ userId }) {
  const records = useContractReadFunction({
    contractAccountId: 'react-near-ts.lantstool.testnet',
    functionName: 'get_records',
    functionArgs: { author_id: userId },
  });

  if (records.isPending) return <p>Loading records...</p>;
  if (records.isError) return <p>Failed to load records</p>;

  return (
    <ul>
      {records.data.result.map((record, i) => (
        <li key={i}>{record}</li>
      ))}
    </ul>
  );
}
```

### With Custom Deserialization

```typescript
import { 
  useContractReadFunction, 
  fromJsonBytes, 
  type DeserializeResultFnArgs 
} from 'react-near-ts';
import * as z from 'zod';

// Define result schema
const ResultSchema = z.array(z.string());

// Custom deserializer with validation
const deserializeResult = (args: DeserializeResultFnArgs) =>
  ResultSchema.parse(fromJsonBytes(args.rawResult));

function ValidatedRecords({ userId }) {
  const records = useContractReadFunction({
    contractAccountId: 'contract.testnet',
    functionName: 'get_records',
    functionArgs: { author_id: userId },
    options: {
      deserializeResult,
    },
  });

  // records.data.result is now typed as string[]
  return (
    <div>
      {records.data?.result.map((record, i) => (
        <p key={i}>{record}</p>
      ))}
    </div>
  );
}
```

### Conditional Querying

```typescript
import { useConnectedAccount, useContractReadFunction } from 'react-near-ts';

function MyRecords() {
  const { connectedAccountId, isConnectedAccount } = useConnectedAccount();

  const records = useContractReadFunction({
    contractAccountId: 'contract.testnet',
    functionName: 'get_records',
    functionArgs: { author_id: connectedAccountId },
    withStateAt: 'LatestOptimisticBlock',
    query: {
      enabled: isConnectedAccount, // Only query when connected
    },
  });

  if (!isConnectedAccount) {
    return <p>Connect wallet to see your records</p>;
  }

  if (records.isPending) return <p>Loading...</p>;
  if (records.isError) return <p>Error loading records</p>;

  return (
    <div>
      <h3>Your Records</h3>
      {records.data.result.map((record, i) => (
        <p key={i}>{record}</p>
      ))}
    </div>
  );
}
```

## Caching Behavior

- **Query Key:** `['callContractReadFunction', contractAccountId, functionName, functionArgs]`
- Automatically caches results based on contract, function, and arguments
- Shares cached data across components with identical parameters
- Automatically refetches on window focus by default (TanStack Query behavior)
- Deduplicates simultaneous identical requests

## Error Handling

The query can fail for several reasons:

- Contract account does not exist
- Function does not exist on contract
- Function execution failed (panic, assertion, etc.)
- Network/RPC errors
- Serialization/deserialization errors

```typescript
if (result.isError) {
  console.error('Contract call error:', result.error);
  // Handle error appropriately
}
```

## Performance Tips

1. **Use conditional querying** to avoid unnecessary requests
2. **Leverage caching** by reusing identical query parameters across components
3. **Use `withStateAt: 'LatestOptimisticBlock'`** for fastest responses (slightly less reliable)
4. **Use `withStateAt: 'LatestFinalBlock'`** for critical data (slower but guaranteed finalized)
