## Overview

`useConnectedAccount` provides access to the currently connected NEAR account. It returns the connected account ID and a boolean flag indicating whether an account is connected.

## Import

```typescript
import { useConnectedAccount } from 'react-near-ts';
```

## Type Signature

```typescript
type UseConnectedAccount = () => {
  connectedAccountId: string | undefined;
  isConnectedAccount: boolean;
};
```

Source: [src/hooks/useConnectedAccount.ts:3](https://github.com/eclipseeer/react-near-ts/blob/main/src/hooks/useConnectedAccount.ts#L3)

## Usage

### Basic Usage

```typescript
import { useConnectedAccount } from 'react-near-ts';

function AccountDisplay() {
  const { connectedAccountId, isConnectedAccount } = useConnectedAccount();

  if (!isConnectedAccount) {
    return <p>No wallet connected</p>;
  }

  return <p>Connected as: {connectedAccountId}</p>;
}
```

### Conditional Rendering

```typescript
import { useConnectedAccount } from 'react-near-ts';

function Dashboard() {
  const { isConnectedAccount } = useConnectedAccount();

  if (!isConnectedAccount) {
    return (
      <div>
        <h2>Welcome</h2>
        <p>Please connect your wallet to continue.</p>
      </div>
    );
  }

  return (
    <div>
      <h2>Dashboard</h2>
      {/* Protected content */}
    </div>
  );
}
```

### With Account Info Hook

```typescript
import { useConnectedAccount, useAccountInfo } from 'react-near-ts';

function AccountBalance() {
  const { connectedAccountId, isConnectedAccount } = useConnectedAccount();
  const accountInfo = useAccountInfo({ accountId: connectedAccountId });

  if (!isConnectedAccount) {
    return <p>Connect wallet to see balance</p>;
  }

  if (accountInfo.isPending) {
    return <p>Loading balance...</p>;
  }

  if (accountInfo.isError) {
    return <p>Failed to load balance</p>;
  }

  return (
    <div>
      <p>Account: {connectedAccountId}</p>
      <p>Balance: {accountInfo.data.accountInfo.balance.total.near} NEAR</p>
    </div>
  );
}
```

### Disabling Actions When Not Connected

```typescript
import { useConnectedAccount, useExecuteTransaction } from 'react-near-ts';
import { transfer } from 'react-near-ts';

function SendTokens() {
  const { isConnectedAccount } = useConnectedAccount();
  const executeTransaction = useExecuteTransaction();

  const handleSend = () => {
    executeTransaction.mutate({
      intent: {
        action: transfer({ amount: { near: '1' } }),
        receiverAccountId: 'receiver.testnet',
      },
    });
  };

  return (
    <button 
      onClick={handleSend}
      disabled={!isConnectedAccount}
    >
      Send NEAR
    </button>
  );
}
```

## Return Value

<ResponseField name="connectedAccountId" type="string | undefined">
  The account ID of the currently connected wallet. Returns `undefined` if no wallet is connected.
  
  Example: `"user.testnet"` or `"alice.near"`
</ResponseField>

<ResponseField name="isConnectedAccount" type="boolean">
  Boolean flag indicating whether a wallet is currently connected.
  
  - `true` if `connectedAccountId` is a string
  - `false` if `connectedAccountId` is `undefined`
</ResponseField>

## Behavior

- The hook subscribes to the NEAR store's `connectedAccountId` state
- Updates automatically when connection state changes
- Returns immediately with current connection status
- No async operations or loading states
